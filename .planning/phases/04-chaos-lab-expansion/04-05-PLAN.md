**Part A: Create ssh-weak Dockerfile and sshd_config**

Create `quantum-chaos-enterprise-lab/ssh/sshd_config`:
```
# /etc/ssh/sshd_config for ssh-weak container (ubuntu:18.04 / OpenSSH 7.6p1)
# Deliberately weak configuration for scanner validation (LAB-05)

Port 22
Protocol 2

# Weak host key types — ssh-audit flags ssh-dss as critical
HostKey /etc/ssh/ssh_host_dsa_key
HostKey /etc/ssh/ssh_host_rsa_key

# Weak KEX algorithms — group1-sha1 and group14-sha1 are critical findings
KexAlgorithms diffie-hellman-group1-sha1,diffie-hellman-group14-sha1,diffie-hellman-group-exchange-sha1

# Weak host key algorithms — ssh-dss is deprecated/critical
HostKeyAlgorithms ssh-dss,ssh-rsa

# Weak MAC algorithms — hmac-md5 and hmac-sha1 are flagged findings
MACs hmac-md5,hmac-sha1,umac-64@openssh.com

PasswordAuthentication yes
PermitRootLogin yes
PermitEmptyPasswords yes
UsePAM no
```

Create `quantum-chaos-enterprise-lab/ssh/Dockerfile`:
```dockerfile
FROM ubuntu:18.04

RUN apt-get update -qq && \
    apt-get install -y --no-install-recommends openssh-server && \
    rm -rf /var/lib/apt/lists/*

# Generate all required host keys including DSA (required for ssh-dss HostKeyAlgorithms)
RUN ssh-keygen -A && \
    ssh-keygen -t dsa -N "" -f /etc/ssh/ssh_host_dsa_key 2>/dev/null || true

# Inject weak sshd_config
COPY sshd_config /etc/ssh/sshd_config

# Create privilege separation dir
RUN mkdir -p /run/sshd

# Create a test user for SSH connections
RUN useradd -m -s /bin/bash labuser && \
    echo "labuser:labpass" | chpasswd

EXPOSE 22

# Run sshd in foreground with debug output for container logs
CMD ["/usr/sbin/sshd", "-D", "-e"]
```

**Part B: Append ssh-weak + ldaps services to docker-compose.yml**

Append the following to `docker-compose.yml` BEFORE the `volumes:` block, AFTER the storage services from Plan 04:

```yaml
  # =========================
  # PHASE 4 — SSH-WEAK PROFILE (profile: ssh-weak)
  # LAB-05: OpenSSH server with deliberately weak KEX/hostkey/MAC algorithms
  # Port: 20022 (safe range — no conflict with existing 2222 ssh-alt)
  # Base: ubuntu:18.04 / OpenSSH 7.6p1 (supports legacy algorithms)
  # =========================
  ssh-weak:
    build:
      context: ./ssh
    profiles: ["ssh-weak"]
    ports:
      - "20022:22"
    restart: unless-stopped

  # =========================
  # PHASE 4 — LDAPS PROFILE (profile: ldaps)
  # LAB-06: OpenLDAP over TLS on standard LDAPS port 636
  # Uses existing lab certs from ./certs/ (modern.crt, modern.key, ca.crt)
  # Independent of identity profile's openldap (port 13890 — unchanged)
  # =========================
  ldaps:
    image: osixia/openldap:1.5.0
    profiles: ["ldaps"]
    environment:
      LDAP_ORGANISATION: "ChaosLab"
      LDAP_DOMAIN: "chaos.local"
      LDAP_ADMIN_PASSWORD: "admin"
      LDAP_TLS: "true"
      LDAP_TLS_CRT_FILENAME: modern.crt
      LDAP_TLS_KEY_FILENAME: modern.key
      LDAP_TLS_CA_CRT_FILENAME: ca.crt
      LDAP_TLS_VERIFY_CLIENT: "never"
    volumes:
      - ./certs:/container/service/slapd/assets/certs:ro
    ports:
      - "636:636"
    restart: unless-stopped
```

Do NOT modify any existing services, especially the `openldap` service in the `identity` profile.
