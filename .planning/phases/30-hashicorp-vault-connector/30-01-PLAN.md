    from cryptography import x509

    from cryptography.hazmat.primitives.asymmetric import rsa


    def _make_test_pem(key_size: int = 2048, hash_alg=None) -> str:
        """Generate a real (but ephemeral) self-signed RSA cert for PKI tests.

        hash_alg defaults to SHA-256; pass hashes.SHA1() to test SHA-1 detection.
        """
        if hash_alg is None:
            hash_alg = hashes.SHA256()
        key = rsa.generate_private_key(public_exponent=65537, key_size=key_size)
        subject = issuer = x509.Name([
            x509.NameAttribute(x509.NameOID.COMMON_NAME, "quirk-test-ca"),
        ])
        cert = (
            x509.CertificateBuilder()
            .subject_name(subject)
            .issuer_name(issuer)
            .public_key(key.public_key())
            .serial_number(x509.random_serial_number())
            .not_valid_before(datetime.now(timezone.utc))
            .not_valid_after(datetime.now(timezone.utc) + timedelta(days=365))
            .sign(key, hash_alg)
        )
        return cert.public_bytes(serialization.Encoding.PEM).decode()


    def _make_mock_client(

        if auth_methods is not None:
            client.sys.list_auth_methods.return_value = {"data": auth_methods}

            client.sys.list_auth_methods.return_value = {"data": {}}
