        # Default: no probe over plaintext.
        if not allow_cleartext:
            # No HTTP probing without explicit opt-in (CR-06).
            return {}

        url = f"http://{host}:{port}/api/overview"
        headers: dict[str, str] = {}
        if credentials:
            user = str(credentials.get("user", "") or credentials.user if hasattr(credentials, "user") else credentials.get("user", ""))

                creds_bytes = base64.b64encode(f"{user}:{password}".encode()).decode()
