       ```python
       from cyclonedx.model.crypto import CryptoPrimitive
       from quirk.cbom.builder import build_cbom
       from quirk.cbom.classifier import classify_algorithm

       def test_no_unknown_classifications_across_lab_profiles():
           seen: dict[str, set[str]] = {}
           for profile, fn in PROFILE_ENDPOINTS.items():
               bom = build_cbom(fn())
               for c in bom.components:
                   cp = getattr(c, "crypto_properties", None)
                   if cp and cp.asset_type and cp.asset_type.value == "algorithm":
                       seen.setdefault(c.name, set()).add(profile)

           unknowns = []
           for name, profiles in sorted(seen.items()):
               primitive, _, _ = classify_algorithm(name)
               # JWT alg:none is an intentional UNKNOWN sentinel (Pitfall #4)
               if primitive == CryptoPrimitive.UNKNOWN and name.lower() != "none":
                   unknowns.append((name, sorted(profiles)))

           assert not unknowns, (
               f"In-scope algorithms classified as UNKNOWN — add rows to "
               f"_ALGORITHM_TABLE in quirk/cbom/classifier.py: {unknowns}"
           )
       ```

       ```python
       import os
       from pathlib import Path

       @pytest.mark.slow
       @pytest.mark.skipif(
           os.environ.get("REGEN_CBOM_COVERAGE") != "1",
           reason="set REGEN_CBOM_COVERAGE=1 to regenerate docs/cbom-classifier-coverage.md",
       )
       def test_regenerate_coverage_report():
           rows = []  # (name, primitive_name, nist_level, classical_bits, profiles_csv)
           seen: dict[str, set[str]] = {}
           for profile, fn in PROFILE_ENDPOINTS.items():
               bom = build_cbom(fn())
               for c in bom.components:
                   cp = getattr(c, "crypto_properties", None)
                   if cp and cp.asset_type and cp.asset_type.value == "algorithm":
                       seen.setdefault(c.name, set()).add(profile)

           for name, profiles in sorted(seen.items()):
               primitive, nist, classical = classify_algorithm(name)
               rows.append((
                   name,
                   primitive.name,
                   "" if nist is None else str(nist),
                   "" if classical is None else str(classical),
                   ", ".join(sorted(profiles)),
               ))

           lines = [
               "# CBOM Classifier Coverage Report",
               "",
               "Generated by `tests/test_cbom_classifier_coverage.py::test_regenerate_coverage_report`.",
               "Run `REGEN_CBOM_COVERAGE=1 pytest tests/test_cbom_classifier_coverage.py::test_regenerate_coverage_report -s` to refresh.",
               "",
               "Per Phase 42 / D-05 — every algorithm surfaced by `build_cbom()` against any chaos lab profile.",
               "",
               "| Algorithm Name | Primitive | NIST Level | Classical Bits | Surfaced By Profiles |",
               "|----------------|-----------|------------|----------------|-----------------------|",

           for name, prim, nist, classical, profiles in rows:

       ```
