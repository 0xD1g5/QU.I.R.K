      from pathlib import Path
      import yaml

      COMPOSE_FILE = Path(__file__).resolve().parent.parent \
          / "quantum-chaos-enterprise-lab" / "docker-compose.yml"

      def _is_pinned(image: str) -> bool:
          if image.endswith(":latest"):
              return False
          if ":" not in image:
              return False
          return True

      def test_every_image_is_pinned():
          data = yaml.safe_load(COMPOSE_FILE.read_text())
          violations = []
          for name, svc in (data.get("services") or {}).items():
              img = svc.get("image")
              if img is None:
                  continue
              if not _is_pinned(img):
                  violations.append(f"{name}: {img}")
          assert not violations, (
              "Unpinned chaos-lab images (CHAOS-05 policy violation):\n  "
              + "\n  ".join(violations)
          )

      import os, platform, subprocess, time
      from pathlib import Path
      import pytest

      LAB_DIR = Path(__file__).resolve().parent.parent / "quantum-chaos-enterprise-lab"
      COMPOSE = LAB_DIR / "docker-compose.yml"

      def _docker_available() -> bool:
          try:
              r = subprocess.run(
                  ["docker", "info"],
                  capture_output=True, timeout=5,
              )
              return r.returncode == 0
          except Exception:
              return False

      pytestmark = pytest.mark.skipif(
          not _docker_available(),
          reason="Docker daemon not available",
      )

      @pytest.fixture(scope="module")
      def profiles():
          r = subprocess.run(
              ["docker", "compose", "-f", str(COMPOSE), "config", "--profiles"],
              capture_output=True, text=True, check=True,
          )
          return sorted({line.strip() for line in r.stdout.splitlines() if line.strip()})

      @pytest.mark.slow
      def test_every_profile_discovered(profiles):
          # Sanity: smime + adcs must be enumerated (CHAOS-06 parity)
          assert "smime" in profiles
          assert "adcs"  in profiles

      def _up(profile: str) -> subprocess.CompletedProcess:
          return subprocess.run(
              ["./lab.sh", "up", "--profile", profile],
              cwd=LAB_DIR, capture_output=True, text=True, timeout=300,
          )

      def _down() -> None:
          subprocess.run(["./lab.sh", "down"], cwd=LAB_DIR,
                         capture_output=True, timeout=120)

      @pytest.mark.slow
      def test_profile_re_up_is_idempotent(profiles):
          # Parametrize manually — pytest's parametrize doesn't see fixtures
          # for the param list. Use a loop with assert + collect errors.
          failures = []
          for profile in profiles:
              if profile == "kerberos" and platform.system() == "Darwin" \
                 and os.environ.get("LAB_INCLUDE_KERBEROS") != "1":
                  continue
              try:
                  r1 = _up(profile)
                  assert r1.returncode == 0, f"{profile} first up failed: {r1.stderr}"
                  time.sleep(15)  # let seed sidecars settle
                  r2 = _up(profile)
                  assert r2.returncode == 0, f"{profile} second up failed: {r2.stderr}"
              except AssertionError as e:
                  failures.append(str(e))
              finally:
                  _down()
          assert not failures, "Idempotency failures:\n  " + "\n  ".join(failures)
