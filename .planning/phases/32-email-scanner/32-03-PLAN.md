**6. Stdlib fallback helpers** — four small functions, each returns `(tls_version, cipher_name, der_bytes)`:

```python
def _fallback_smtp_starttls(host, port, timeout, ctx):
    with smtplib.SMTP(host, port, timeout=timeout) as smtp:
        smtp.ehlo()
        smtp.starttls(context=ctx)
        smtp.ehlo()
        ssock = smtp.sock
        if not isinstance(ssock, ssl.SSLSocket):
            ssock = getattr(getattr(smtp, "file", None), "_sock", None)  # Pitfall 2
        if not isinstance(ssock, ssl.SSLSocket):
            raise RuntimeError("smtplib STARTTLS did not yield ssl.SSLSocket")
        return ssock.version(), (ssock.cipher() or (None,))[0], ssock.getpeercert(binary_form=True)


def _fallback_imap_starttls(host, port, timeout, ctx):
    imap = imaplib.IMAP4(host, port, timeout=timeout)
    try:
        imap.starttls(ssl_context=ctx)
        ssock = getattr(imap, "sock", None)
        if not isinstance(ssock, ssl.SSLSocket):
            raise RuntimeError("imaplib STARTTLS did not yield ssl.SSLSocket")
        return ssock.version(), (ssock.cipher() or (None,))[0], ssock.getpeercert(binary_form=True)
    finally:
        try:
            imap.logout()
        except Exception:
            pass


def _fallback_pop3_starttls(host, port, timeout, ctx):
    pop = poplib.POP3(host, port=port, timeout=timeout)
    try:
        pop.stls(context=ctx)
        ssock = getattr(pop, "sock", None)
        if not isinstance(ssock, ssl.SSLSocket):
            raise RuntimeError("poplib STARTTLS did not yield ssl.SSLSocket")
        return ssock.version(), (ssock.cipher() or (None,))[0], ssock.getpeercert(binary_form=True)
    finally:
        try:
            pop.quit()
        except Exception:
            pass


def _fallback_implicit_tls(host, port, timeout, ctx):
    """SMTPS/IMAPS/POP3S — direct TLS handshake on connect."""
    sock = socket.create_connection((host, port), timeout=timeout)
    try:
        ssock = ctx.wrap_socket(sock, server_hostname=host)
        return ssock.version(), (ssock.cipher() or (None,))[0], ssock.getpeercert(binary_form=True)
    finally:
        try:
            sock.close()
        except Exception:
            pass
```

```python
def scan_one_email(host, port, protocol_label, starttls_enum, timeout, logger=None, session_start=None):
    ep = _scan_one_sslyze_email(host, port, starttls_enum, timeout, logger)
    if ep is None:
        ep = _scan_one_fallback_email(host, port, protocol_label, timeout, logger)
    ep.protocol = protocol_label
    ep.service_detail = f"{protocol_label}:{port}"
    ep.scanned_at = (session_start or datetime.now(timezone.utc)).replace(tzinfo=None)
    return ep
```
