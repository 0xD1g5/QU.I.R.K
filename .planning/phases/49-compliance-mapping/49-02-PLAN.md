    ```python
    def status_report(format: str = "text") -> None:
        """Print per-framework version + last_verified + source_url to stdout.
        Used by `quirk compliance status` (D-05, COMPLY-09)."""
        # Group entries by framework, deduping (framework, version, last_verified, source_url).
        seen: Dict[str, Dict[str, Any]] = {}
        for entries in COMPLIANCE_MAP.values():
            for e in entries:
                key = e["framework"]
                # Keep the OLDEST last_verified per framework (worst-case staleness signal).
                if key not in seen or e["last_verified"] < seen[key]["last_verified"]:
                    seen[key] = {
                        "framework": e["framework"], "version": e["version"],
                        "last_verified": e["last_verified"], "source_url": e["source_url"],
                    }
        if format == "json":
            import json as _json
            print(_json.dumps(seen, indent=2, sort_keys=True))
            return
        # text default — fixed-width-ish ASCII
        print(f"{'Framework':<20} {'Version':<14} {'Last Verified':<14} Source URL")
        print("-" * 100)
        for fw in sorted(seen):
            row = seen[fw]
            print(f"{row['framework']:<20} {row['version']:<14} {row['last_verified']:<14} {row['source_url']}")
    ```
