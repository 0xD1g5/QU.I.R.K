def collect_emitted_titles() -> set[str]:
    """Return the set of all FIXED-STRING finding titles literally emitted by
    quirk.engine.risk_engine, plus the NORMALIZED form of every f-string title.
    Source-of-truth strategy: parse risk_engine.py via ast and walk Call nodes
    where func.id == '_build_finding', extracting the `title=` keyword value.
    For ast.JoinedStr (f-strings) build the literal-only template and apply the

        re.sub(r'\\s*\\([^)]+\\)\\s*$', '', literal).strip()

    """

       ```python
       _PAREN_SUFFIX_RE = re.compile(r"\s*\([^)]+\)\s*$")
       _RISK_ENGINE = pathlib.Path(__file__).resolve().parents[2] / "quirk/engine/risk_engine.py"

       def _normalize(title: str) -> str:
           return _PAREN_SUFFIX_RE.sub("", title).strip()

       def collect_emitted_titles() -> set[str]:
           tree = ast.parse(_RISK_ENGINE.read_text())
           titles: set[str] = set()
           for node in ast.walk(tree):
               if not (isinstance(node, ast.Call) and isinstance(node.func, ast.Name)
                       and node.func.id == "_build_finding"):
                   continue
               for kw in node.keywords:
                   if kw.arg != "title":
                       continue
                   v = kw.value
                   if isinstance(v, ast.Constant) and isinstance(v.value, str):
                       titles.add(_normalize(v.value))
                   elif isinstance(v, ast.JoinedStr):
                       # f-string: build literal-only template (drop FormattedValue parts entirely)
                       lit = "".join(p.value for p in v.values if isinstance(p, ast.Constant))
                       titles.add(_normalize(lit))
           return titles
       ```
