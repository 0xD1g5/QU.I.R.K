    ```python
    def _wrapped_phase(run_stats, phase_name, scanner_label, fn, error_endpoints, logger):
        """D-14: run a scanner phase with BaseException protection.

        Returns the scanner's return value, or [] on captured exception.
        Re-raises only KeyboardInterrupt and SystemExit.
        """
        try:
            with _phase_timer(run_stats, phase_name):
                return fn()
        except (KeyboardInterrupt, SystemExit):
            raise
        except BaseException as exc:
            logger.error(f"{phase_name}: unhandled exception: {exc!r}")
            from quirk.models import CryptoEndpoint
            error_endpoints.append(CryptoEndpoint(
                host=scanner_label, port=0, protocol="ERROR",
                scan_error=str(exc),
                scan_error_category="exception",
            ))
            return []
    ```
