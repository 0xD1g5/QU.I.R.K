       def _dispatch_schedule(schedule: ScheduledScan, db: Session, config_path: str) -> ScheduledRun:
           now = _utcnow_naive()
           run = ScheduledRun(schedule_id=schedule.id, dispatched_at=now,
                              status="pending", scan_output_path=None, scan_id=None)
           db.add(run); db.flush()  # get run.id
           output_dir = Path("output/scheduled") / schedule.name / now.strftime("%Y%m%d-%H%M%S")
           output_dir.mkdir(parents=True, exist_ok=True)
           cmd = [sys.executable, "-m", "run_scan",
                  "--config", config_path,
                  "--target", schedule.target,
                  "--profile", schedule.profile or "balanced",
                  "--output", str(output_dir)]
           run.status = "running"
           db.commit()
           try:
               proc = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
               stdout, stderr = proc.communicate()
               run.status = "completed" if proc.returncode == 0 else "failed"
           except Exception:
               run.status = "failed"
           run.completed_at = _utcnow_naive()
           run.scan_output_path = str(output_dir)
           schedule.last_run_at = now
           db.commit()
           return run
       ```

       def run_scheduler(argv: list[str]) -> None:
           parser = argparse.ArgumentParser(prog="quirk scheduler")
           sub = parser.add_subparsers(dest="action", required=True)
           p_run = sub.add_parser("run")
           p_run.add_argument("--config", default=None)
           p_run.add_argument("--once", action="store_true",
                              help="Run a single iteration and exit (for tests).")
           args = parser.parse_args(argv)
           db_path = _resolve_db_path(args.config)

           signal.signal(signal.SIGINT, _handle_signal)
           signal.signal(signal.SIGTERM, _handle_signal)

           with get_session(db_path) as db:
               _recover_stale_runs(db)

           if args.once:
               with get_session(db_path) as db:
                   _check_and_dispatch_due(db, db_path)
               return

               with get_session(db_path) as db:
                   _check_and_dispatch_due(db, db_path)
               for _ in range(60):
                   if _stop_flag:
                       break
                   time.sleep(1)
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Scheduler process → child quirk subprocess | Schedule fields (`name`, `target`, `profile`) flow into argv of a child Python process |
| Scheduler process → SQLite | Concurrent writes from scheduler + dashboard API (Plan 03) require lock tolerance |
| OS signals → scheduler process | SIGINT/SIGTERM must shut down cleanly without leaving "running" rows |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
