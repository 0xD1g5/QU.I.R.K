```python
def _make_ep(
    db,
    host: str,
    port: int,
    protocol: str,
    severity: str,
    scanned_at: Optional[datetime] = None,
    scan_error: Optional[str] = None,
) -> CryptoEndpoint:

        protocol=protocol,

        scan_error=scan_error,
    )
    db.add(ep)
    db.commit()
    return ep
```
