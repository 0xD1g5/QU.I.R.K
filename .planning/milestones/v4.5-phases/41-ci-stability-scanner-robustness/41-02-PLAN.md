    def test_deprecated_timeout_seconds_alias_warns():
        cfg = config_from_dict({"scan": {"concurrency": 4, "timeouts": {"default_seconds": 9}}})

            v = cfg.scan.timeout_seconds
            assert v == 9
            assert any(issubclass(x.category, DeprecationWarning) for x in w)

        cfg = config_from_dict({"scan": {"concurrency": 4, "timeouts": {"tls_seconds": 6, "ssh_seconds": 7, "fingerprint_seconds": 3}}})
