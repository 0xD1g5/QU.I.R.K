  <action>
    Create `tests/test_adcs_ast_gate.py` by cloning the structural shape of

    with the three forbidden-set walkers per D-80-R4.

    Body sketch:
    ```
    import ast, pathlib
    PROJECT_ROOT = pathlib.Path(__file__).resolve().parent.parent
    TARGET = PROJECT_ROOT / "quirk" / "scanner" / "adcs_scanner.py"

    FORBIDDEN_IMPORT_MODULES = {"certipy_ad", "certipy", "impacket.ldap.ldapasn1_modify"}
    FORBIDDEN_FROM_NAMES = {("cryptography.x509", "CertificateSigningRequestBuilder")}
    FORBIDDEN_LDAP_METHODS = {"add", "modify", "delete", "modify_dn"}

    def _walk(source: str) -> list[str]:
        violations: list[str] = []
        tree = ast.parse(source)
        for node in ast.walk(tree):
            # 1. Module-level imports
            if isinstance(node, ast.Import):
                for alias in node.names:
                    if alias.name in FORBIDDEN_IMPORT_MODULES \
                       or alias.name.startswith("certipy"):
                        violations.append(f"import {alias.name}")
            # 2. from-imports: name-level check
            elif isinstance(node, ast.ImportFrom):
                mod = node.module or ""
                if mod in FORBIDDEN_IMPORT_MODULES or mod.startswith("certipy"):
                    violations.append(f"from {mod} import ...")
                for alias in node.names:
                    if (mod, alias.name) in FORBIDDEN_FROM_NAMES:
                        violations.append(f"from {mod} import {alias.name}")
            # 3. Attribute access: forbid <any>.add / .modify / .delete / .modify_dn
            elif isinstance(node, ast.Attribute):
                if node.attr in FORBIDDEN_LDAP_METHODS:
                    # only count if it looks like a method call site: parent is
                    # an ast.Call with this Attribute as .func. Conservative.
                    violations.append(f".{node.attr}() attribute access")
        return violations

    def test_real_target_is_clean():
        violations = _walk(TARGET.read_text())
        # The .add/.modify walker is conservative — if real source legitimately
        # uses .add/.modify outside ldap3 context, this test may need refinement.
        # Plan 80-02 was written specifically to avoid those method names.
        assert not violations, f"ADCS-09 violation in adcs_scanner.py: {violations}"

    def test_synthetic_forbidden_fails():
        bad = "import certipy_ad\nfrom cryptography.x509 import CertificateSigningRequestBuilder\nconn.add(x)\n"
        assert _walk(bad), "AST gate should flag synthetic forbidden source"

    def test_synthetic_clean_passes():
        good = "import ldap3\nfrom quirk.util.weak_crypto import is_weak_cipher\nconn.bind()\nconn.search(x)\n"
        assert not _walk(good), "AST gate should accept synthetic clean source"
    ```

    Notes for the executor:
    - The `.add/.modify/.delete/.modify_dn` walker per D-80-R4 is broad — if
      `quirk/scanner/adcs_scanner.py` legitimately needs a `.add()` call for
      a NON-ldap3 object (e.g., `list.add` — Python lists use `.append`, so
      this should not arise), refine the walker to gate on the receiver being
      `conn`/`Connection`. Plan 80-02 was authored to avoid these method names
      entirely, so the broad walker should be clean.
    - The three-test shape (real target / synthetic forbidden / synthetic
      clean) mirrors `test_smime_ast_gate.py` Phase 79 contract.

    Implements ADCS-09 (AST gate leg).
