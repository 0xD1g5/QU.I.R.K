    - `msPKI-Enrollment-Flag` (Integer)
    - `msPKI-RA-Signature` (Integer)

    `pKICertificateTemplate` listing those attrs as MAY.

    ```yaml

      image: bitnamilegacy/openldap:2.6.10-debian-12-r4
      profiles: ["adcs"]
      environment:
        LDAP_ROOT: "dc=quirk,dc=lab"
        LDAP_ADMIN_USERNAME: "admin"
        LDAP_ADMIN_PASSWORD: "admin"
        LDAP_PORT_NUMBER: 389
        LDAP_ALLOW_ANON_BINDING: "yes"   # chaos lab only; real AD rejects this per D-80-R3
      volumes:
        - ./adcs/schema:/schema:ro
        - ./adcs/ldif:/ldif:ro
        - ./adcs/certs:/adcs-certs:ro
      ports:
        - "38910:389"
      restart: unless-stopped

    adcs-seed:
      image: bitnamilegacy/openldap:2.6.10-debian-12-r4
      profiles: ["adcs"]
      depends_on:
        adcs-openldap:
          condition: service_started
      entrypoint: ["/bin/sh", "-c"]
      command:
        - >-
          sleep 8 &&
          ldapadd -c -Y EXTERNAL -H ldap://adcs-openldap:389 -f /schema/msPKI-schema.ldif || true ;
          ldapadd -c -x -H ldap://adcs-openldap:389
            -D 'cn=admin,dc=quirk,dc=lab' -w admin -f /ldif/00-base.ldif ;
          ldapadd -c -x -H ldap://adcs-openldap:389
            -D 'cn=admin,dc=quirk,dc=lab' -w admin -f /ldif/10-ca.ldif ;

            -D 'cn=admin,dc=quirk,dc=lab' -w admin -f /ldif/20-templates.ldif

        - ./adcs/ldif:/ldif:ro
    ```
