**tests/test_cmvp_no_certified_true.py** — clone the AST walker from
tests/test_smime_ast_gate.py:28-46. Per D-81-R7 + RESEARCH §AST Invariant Test,
implement THREE detection patterns (each test name should reflect which one):

```python
import ast, pathlib, pytest

PROJECT_ROOT = pathlib.Path(__file__).resolve().parents[1]
FORBIDDEN_KEY = "certified"
TARGET_DIRS = [PROJECT_ROOT / "quirk" / "compliance",
               PROJECT_ROOT / "quirk" / "cbom"]

def _iter_py_files():
    for d in TARGET_DIRS:
        yield from d.rglob("*.py")

def _collect_violations(tree, src_path):
    violations = []
    for node in ast.walk(tree):
        # Pattern 1: dict literal {"certified": True}
        if isinstance(node, ast.Dict):
            for k, v in zip(node.keys, node.values):
                if (isinstance(k, ast.Constant) and k.value == FORBIDDEN_KEY
                    and isinstance(v, ast.Constant) and v.value is True):
                    violations.append(f"{src_path}:{k.lineno}: dict literal 'certified': True")
        # Pattern 2: kwarg certified=True on any Call
        if isinstance(node, ast.Call):
            for kw in node.keywords:
                if (kw.arg == FORBIDDEN_KEY
                    and isinstance(kw.value, ast.Constant) and kw.value.value is True):
                    violations.append(f"{src_path}:{node.lineno}: kwarg certified=True")
        # Pattern 3: subscript/attr assignment d["certified"] = True / obj.certified = True
        if isinstance(node, ast.Assign):
            for tgt in node.targets:
                if (isinstance(tgt, ast.Subscript) and isinstance(tgt.slice, ast.Constant)
                    and tgt.slice.value == FORBIDDEN_KEY
                    and isinstance(node.value, ast.Constant) and node.value.value is True):
                    violations.append(f"{src_path}:{node.lineno}: subscript ['certified'] = True")
                if (isinstance(tgt, ast.Attribute) and tgt.attr == FORBIDDEN_KEY
                    and isinstance(node.value, ast.Constant) and node.value.value is True):
                    violations.append(f"{src_path}:{node.lineno}: attribute .certified = True")
    return violations

def test_no_certified_true_in_cmvp_or_cbom():
    all_violations = []
    for src in _iter_py_files():
        tree = ast.parse(src.read_text())
        all_violations.extend(_collect_violations(tree, src))
    assert not all_violations, (
        "v4.10-D-01 / CMVP-07 PERMANENT INVARIANT violated.\n"
        + "\n".join(all_violations)
    )
```
