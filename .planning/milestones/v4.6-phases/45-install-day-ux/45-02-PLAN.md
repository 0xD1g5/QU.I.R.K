    ```python
    """Phase 45: centralized optional-extra registry + probe.

    Per Phase 45 / D-08, D-10, and Q1/Q3 user decisions:
    - REGISTRY covers: identity, db, cloud, dashboard.
    - motion is INTENTIONALLY OMITTED — Phase 41 inline _emit_missing_extra_advisory calls

      double-emit and would never fire because no enable_* flag is dedicated to it.
    - D-11: existing per-scanner `*_AVAILABLE` flags (broker_scanner.SSLYZE_AVAILABLE,
      vault_connector.HVAC_AVAILABLE, etc.) are NOT migrated — they keep their existing
      patch points for the 9+ test files that depend on them.
    """
    from __future__ import annotations
    from dataclasses import dataclass
    from importlib.util import find_spec
    from typing import Tuple

    @dataclass(frozen=True)
    class OptionalExtra:
        extra: str
        modules: Tuple[str, ...]
        scanner_label: str
        install_hint: str
        enabled_attrs: Tuple[str, ...]

    REGISTRY: Tuple[OptionalExtra, ...] = (
        OptionalExtra(
            extra="identity",
            modules=("impacket",),
            scanner_label="kerberos_scanner",
            install_hint="Kerberos scanning skipped — run `pip install quirk[identity]` to enable",
            enabled_attrs=("enable_kerberos",),
        ),
        OptionalExtra(
            extra="db",
            modules=("psycopg2", "pymysql"),
            scanner_label="db_connector",
            install_hint="Database TLS scanning (PostgreSQL/MySQL) skipped — run `pip install quirk[db]` to enable",
            enabled_attrs=("enable_db",),
        ),
        OptionalExtra(
            extra="cloud",
            modules=("googleapiclient", "kubernetes", "hvac"),
            scanner_label="cloud_connectors",
            install_hint="GCP / Kubernetes / HashiCorp Vault scanning skipped — run `pip install quirk[cloud]` to enable",
            enabled_attrs=("enable_gcp", "enable_k8s", "enable_vault"),
        ),
        OptionalExtra(
            extra="dashboard",
            modules=("fastapi", "uvicorn", "playwright"),
            scanner_label="dashboard",
            install_hint="Web dashboard / PDF export unavailable — run `pip install quirk[dashboard]` to enable",
            enabled_attrs=("enable_dashboard_pdf",),  # see step 3 below
        ),
    )

    def is_extra_available(extra: str) -> bool:
        entry = next((e for e in REGISTRY if e.extra == extra), None)
        if entry is None:
            return False
        return all(find_spec(m) is not None for m in entry.modules)

    def probe_missing_extras(cfg, error_endpoints) -> None:
        """For each registry entry whose enable_* flag is True AND whose modules are
        unimportable, append one ADVISORY CryptoEndpoint row to error_endpoints.

        D-08: config-disabled scanners stay silent.
        D-05: one advisory per skipped scanner — never aggregated.
        """
        from quirk.models import CryptoEndpoint  # local import to avoid circulars
        for entry in REGISTRY:
            if entry.enabled_attrs:
                if not any(getattr(getattr(cfg, "connectors", cfg), a, False)
                           for a in entry.enabled_attrs):
                    continue  # all gating flags False → skip silently
            if all(find_spec(m) is not None for m in entry.modules):
                continue  # extra is available → no advisory
            error_endpoints.append(CryptoEndpoint(
                host=entry.scanner_label,
                port=0,
                protocol="ADVISORY",
                scan_error=entry.install_hint,
                scan_error_category="missing_extra",
            ))
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user config → registry walk | `cfg.connectors.enable_*` values cross into the probe; only attribute reads (no eval/exec). |
| Python import system → probe | `importlib.util.find_spec` is invoked on a static, module-level allow-list of names; no user input controls which modules are probed. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-45-04 | Tampering | install_hint string | mitigate | Hint strings are module-level constants on a frozen dataclass; never f-string-interpolated from runtime data (RESEARCH.md security §V5). |
| T-45-05 | DoS | find_spec on a non-existent module | accept | find_spec returns None for unknown modules in O(1)-ish stat time; bounded by registry size (4 entries). |
| T-45-06 | Information Disclosure | advisory message includes module names | accept | Names are well-known PyPI packages (impacket, psycopg2, etc.); no secrets. |
| T-45-07 | Tampering | Partial-install ImportError on probe | mitigate | Use `find_spec` (does not import); cannot trigger ImportError on partial installs (RESEARCH.md anti-pattern). |
</threat_model>
