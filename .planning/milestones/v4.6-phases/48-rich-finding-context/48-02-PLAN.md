    ```python
    import json
    import os
    import tempfile
    from types import SimpleNamespace

    from quirk.reports.writer import write_reports


    def _cfg(outdir):
        return SimpleNamespace(
            output=SimpleNamespace(directory=outdir),
            scan=SimpleNamespace(),
        )


    def _findings_fixture():
        return [
            {
                "severity": "HIGH",
                "host": "10.0.0.1",
                "port": 443,
                "title": "RSA certificate quantum-vulnerable",
                "description": "This certificate uses RSA, vulnerable to a sufficiently large quantum computer.",
                "recommendation": (
                    "Plan migration to ML-KEM (FIPS 203) and ML-DSA (FIPS 204). "
                    "Per NIST IR 8547, RSA and ECC are deprecated after 2030 and disallowed after 2035."
                ),
            },
            {
                "severity": "LOW",
                "host": "10.0.0.2",
                "port": 443,
                "title": "Legacy TLS cipher suites accepted",
                "description": "This service accepts TLS 1.0/1.1 cipher suites that are deprecated and exploitable.",
                "recommendation": "Disable TLS 1.0/1.1 and restrict to TLS 1.2+ AEAD cipher suites.",
            },
        ]


    def test_json_export_preserves_description():
        with tempfile.TemporaryDirectory() as outdir:
            write_reports(_cfg(outdir), endpoints=[], findings=_findings_fixture())
            json_files = [f for f in os.listdir(outdir) if f.startswith("findings-") and f.endswith(".json")]
            assert json_files, "No findings JSON written"
            data = json.loads(open(os.path.join(outdir, json_files[0])).read())
            assert all("description" in f for f in data), f"description missing in {data}"
            assert all(f["description"].strip() for f in data), "Empty description in JSON export"


    def test_json_export_preserves_deprecation_phrase():
        with tempfile.TemporaryDirectory() as outdir:
            write_reports(_cfg(outdir), endpoints=[], findings=_findings_fixture())
            json_files = [f for f in os.listdir(outdir) if f.startswith("findings-") and f.endswith(".json")]
            data = json.loads(open(os.path.join(outdir, json_files[0])).read())
            qv = next(f for f in data if "RSA" in f["title"])
            assert "Per NIST IR 8547" in qv["recommendation"]
            assert "FIPS 203" in qv["recommendation"]


    def test_html_report_has_description_column():
        with tempfile.TemporaryDirectory() as outdir:
            write_reports(_cfg(outdir), endpoints=[], findings=_findings_fixture())
            html_files = [f for f in os.listdir(outdir) if f.startswith("report-") and f.endswith(".html")]
            assert html_files, "No HTML report written"
            html = open(os.path.join(outdir, html_files[0])).read()
            assert html.count("<th>Description</th>") >= 2, "Description column missing in HTML All Findings"
    ```
