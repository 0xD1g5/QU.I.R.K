```
def _prompt_int(prompt: str, default: int, minv: int, maxv: int) -> int:
    if not (minv <= default <= maxv):
        raise ValueError(f"_prompt_int default {default} out of range [{minv}, {maxv}]")
    for _ in range(3):  # bound the retry loop
        try:
            raw = _prompt(prompt, str(default))
        except (EOFError, KeyboardInterrupt):
            return default
        try:
            value = int(raw)
        except ValueError:
            print(f"Invalid integer {raw!r}; expected an int in [{minv}, {maxv}].")
            continue
        if minv <= value <= maxv:
            return value
        print(f"Value {value} out of range [{minv}, {maxv}].")
    return default  # fall through to default after 3 retries (preserves existing caller contract)
```
