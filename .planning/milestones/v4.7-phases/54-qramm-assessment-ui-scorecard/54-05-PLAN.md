    2. **Compute completion percentages from context** (per dimension):
       ```tsx
       const completionByDim = useMemo(() => {
         const out: Record<string, number> = {}
         for (const dim of DIMENSIONS) {
           const areas = DIMENSION_PRACTICE_AREAS[dim]
           // 30 questions per dimension (3 areas × 10)
           let answered = 0
           for (const [qn, a] of ctx.answers) {
             if (a.answer_value == null) continue
             // Find practice_area for question via DIMENSION_PRACTICE_AREAS — use the question_number→area mapping that we'd need.
             // Simpler: dimension is encoded by question_number range — questions 1-30 are CVI, 31-60 SGRM, 61-90 DPE, 91-120 ITR.
             const dimIdx = Math.floor((qn - 1) / 30)
             if (DIMENSIONS[dimIdx] === dim) answered += 1
           }
           out[dim] = Math.round((answered / 30) * 100)
         }
         return out
       }, [ctx.answers])
       ```

    3. **Maturity distribution** (count practice areas at each maturity 1-4 — use the dimensions[].score from scoreResult, rounded to nearest int):
       ```tsx
       const maturityDist = useMemo(() => {
         const dist = { 1: 0, 2: 0, 3: 0, 4: 0 } as Record<number, number>
         if (!ctx.scoreResult) return dist
         // For each of 12 practice areas (4 dims × 3 areas), we'd need per-area scores — but the score endpoint
         // returns per-dimension only. Approximate: bucket each dimension's score (rounded) and count.
         for (const dim of DIMENSIONS) {
           const dscore = ctx.scoreResult.dimensions[dim]?.score
           if (dscore == null) continue
           const bucket = Math.max(1, Math.min(4, Math.round(dscore)))
           dist[bucket] += 1
         }
         return dist
       }, [ctx.scoreResult])
       ```

    4. **Calculate Score handler:**
       ```tsx
       async function handleCalculate() {
         if (!ctx.sessionId) return
         setCalculating(true)
         setError(null)
         try {
           const resp = await fetch(`/api/qramm/sessions/${ctx.sessionId}/score`, {
             method: "POST",
             headers: { "Content-Type": "application/json" },
             body: JSON.stringify({
               profile_multiplier: ctx.profile?.multiplier ?? null,
             }),
           })
           if (!resp.ok) {
             setError("Could not calculate score — check your connection and try again")
             return
           }
           const json: QRAMMScoreResponse = await resp.json()
           ctx.setScoreResult(json)
         } catch {
           setError("Could not calculate score — check your connection and try again")
         } finally {
           setCalculating(false)
         }
       }
       ```
